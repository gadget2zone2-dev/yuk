```python
import os
import zipfile

def create_project_files():
    # Create directories
    os.makedirs('templates', exist_ok=True)
    
    # File contents (copy from above)
    files = {
        'app.py': '''[Content from app.py above]''',
        'requirements.txt': '''[Content from requirements.txt above]''',
        '.env': '''[Content from .env above]''',
        'templates/index.html': '''[Content from index.html above]''',
        '.gitignore': '''[Content from .gitignore above]''',
        'README.md': '''[Content from README.md above]''',
        'Procfile': 'web: gunicorn app:app'
    }
    
    for filepath, content in files.items():
        with open(filepath, 'w', encoding='utf-8') as f:
            f.write(content)
    
    # Create ZIP
    with zipfile.ZipFile('ai-content-studio.zip', 'w') as zipf:
        for root, dirs, files in os.walk('.'):
            for file in files:
                if file != 'create_project.py' and not file.endswith('.zip'):
                    zipf.write(os.path.join(root, file))
    
    print("âœ… Project files created and zipped as 'ai-content-studio.zip'")

if __name__ == '__main__':
    create_project_files()
